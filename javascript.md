### Objects

  - Add comas at the beginning of new line

```js
      
// Good example
var foo = {
        age: 30
    ,   name: 'John'
    ,   ocupation: 'Economist'
}

// Bad example
var foo = {
    age: 30,
    name: 'John',
    ocupation: 'Economist'
}
```

### Arrays

   - Add comas at the beginning of new line
    
```js

// Good example
var foo = [
        'red', 'green', 'blue', 'yellow'
    ,   'black', 'white'
]

// Bad example
var foo = [
        'red', 'green', 'blue', 'yellow',
        'black', 'white'
]
```

## Naming Conventions

   - Avoid single letter name, be descriptive.
   
```js

// Bad example
function q() {
  // ...stuff...
}

// Good example
function query() {
  // ..stuff..
}
```
